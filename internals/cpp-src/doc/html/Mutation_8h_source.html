209  std::string classify(const std::string &local_target,
210  const int target_pos) const {
211  int d = right - left - 1;
212  std::string s;
213  try {
214  if (d == 1 and seq.length() == 0) {
215  // deletion of a single nucleotide
216  int i = left + 1 - target_pos;
217  s = local_target.substr(left + 1 - target_pos, 1) + '-';
218  } else if (d == 0 and seq.length() == 1) {
219  // insertion of a single nucleotide
220  s = '-' + seq;
221  } else if (d == 1 and seq.length() == 1) {
222  // single-nucleotide mismatch
223  if (seq == "N") {
224  s = "N_match";
225  } else {
226  s = local_target[left + 1 - target_pos] + seq;
227  }
228  } else if (d > 1 and seq.length() == 0) {
229  s = "multinuc_deletion";
230  } else if (d == 0 and seq.length() > 1) {
231  s = "multinuc_insertion";
232  } else if (d == seq.length()) {
233  if (local_target.substr(right - 1 - target_pos, 1) == "A") {
234  s = "A_multinuc_mismatch";
235  } else if (local_target.substr(right - 1 - target_pos, 1) == "C") {
236  s = "C_multinuc_mismatch";
237  } else if (local_target.substr(right - 1 - target_pos, 1) == "G") {
238  s = "G_multinuc_mismatch";
239  } else if (local_target.substr(right - 1 - target_pos, 1) == "T") {
240  s = "T_multinuc_mismatch";
241  } else {
242  s = "N_multinuc_mismatch";
243  }
244  } else if (seq.length() < d) {
245  s = "complex_deletion";
246  } else if (seq.length() > d) {
247  s = "complex_insertion";
248  } else {
249  throw std::runtime_error("Error: Unknown error. Unable to classify mutation.");
250  }
251  } catch (std::out_of_range &err) {
252  throw std::runtime_error(
253  "Error: Unable to classify mutation. Mutation location falls outside local target sequence.");
254  }
255  return s;
256  }
